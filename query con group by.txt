QUERY CON GROUP BY
 

Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(`id`) AS `number_of_students`, YEAR(enrolment_date) AS `year` 
FROM `students` 
GROUP BY YEAR(enrolment_date);



2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio


SELECT COUNT(id) AS `number_of_teachers`, `office_address` 
FROM `teachers` 
GROUP BY `office_address`;

SELECT AVG(vote) AS `avarage_vote`, `exam_id` AS `exam`

4. Contare quanti corsi di laurea ci sono per ogni dipartimento


SELECT COUNT(`id`) AS `num_cdl`, `department_id` AS `department` 
FROM `degrees`
